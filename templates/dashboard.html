function loadPreset(key) {
  const p = PRESETS[key];
  if (!p) return;
  document.getElementById('sim-ip').value     = p.ip;
  document.getElementById('sim-method').value = p.method;
  document.getElementById('sim-path').value   = p.path;
  document.getElementById('sim-params').value = p.params;
  document.getElementById('sim-body').value   = p.body;
  clearSimResult();
}

function parseParams(raw) {
  const obj = {};
  if (!raw.trim()) return obj;
  raw.split('\n').forEach(line => {
    const idx = line.indexOf('=');
    if (idx > 0) {
      obj[line.slice(0,idx).trim()] = line.slice(idx+1).trim();
    }
  });
  return obj;
}

async function runSimulation(customIp) {
  const ip     = customIp || document.getElementById('sim-ip').value.trim() || '10.0.0.1';
  const method = document.getElementById('sim-method').value;
  const path   = document.getElementById('sim-path').value.trim() || '/';
  const params = parseParams(document.getElementById('sim-params').value);
  const body   = document.getElementById('sim-body').value.trim();

  const res = await fetch('/simulate', {
    method: 'POST',
    headers: {'Content-Type':'application/json'},
    body: JSON.stringify({ip, method, path, params, headers:{}, body})
  });
  const data = await res.json();

  const el     = document.getElementById('sim-result');
  const header = document.getElementById('sim-result-header');
  const json   = document.getElementById('sim-result-json');

  el.className = 'sim-result show ' + (data.allowed ? 'allowed' : 'blocked');
  header.className = 'result-header ' + (data.allowed ? 'green' : 'red');
  header.textContent = data.allowed ? '✅ ALLOWED — Request passed inspection' : '🚫 BLOCKED — Malicious request detected';
  json.textContent = JSON.stringify(data, null, 2);

  setTimeout(refreshAll, 300);
}

function clearSimResult() {
  const el = document.getElementById('sim-result');
  el.className = 'sim-result';
  el.classList.remove('show');
}
