function loadPreset(key) {
  const p = PRESETS[key]; if (!p) return;
  document.getElementById('sim-ip').value     = p.ip;
  document.getElementById('sim-method').value = p.method;
  document.getElementById('sim-path').value   = p.path;
  document.getElementById('sim-params').value = p.params;
  document.getElementById('sim-body').value   = p.body;
  clearSimResult();
}

function parseParams(raw) {
  const obj = {};
  if (!raw.trim()) return obj;
  raw.split('\n').forEach(line => {
    const idx = line.indexOf('=');
    if (idx > 0) obj[line.slice(0,idx).trim()] = line.slice(idx+1).trim();
  });
  return obj;
}

function runSimulation() {
  const ip     = document.getElementById('sim-ip').value.trim() || '10.0.0.1';
  const method = document.getElementById('sim-method').value;
  const path   = document.getElementById('sim-path').value.trim() || '/';
  const params = parseParams(document.getElementById('sim-params').value);
  const body   = document.getElementById('sim-body').value.trim();

  const result = inspectRequest(ip, method, path, params, body);
  addLog(result);

  const el     = document.getElementById('sim-result');
  const header = document.getElementById('sim-result-header');
  const jsonEl = document.getElementById('sim-result-json');
  el.className = 'sim-result show ' + (result.allowed ? 'allowed' : 'blocked');
  header.className = 'result-header ' + (result.allowed ? 'green' : 'red');
  header.textContent = result.allowed
    ? '✅ ALLOWED — Request passed inspection'
    : '🚫 BLOCKED — Malicious request detected';
  jsonEl.textContent = JSON.stringify(result, null, 2);
  refreshAll();
}

function clearSimResult() {
  document.getElementById('sim-result').className = 'sim-result';
}
